void repeat_det_by_minor(size_t repeat, size_t size)
{   double *a;
    a = (double*) malloc( (size * size) * sizeof(double) );

    while(repeat--)
    {   uniform_01(size * size, a);
        det_by_minor(a, size);
    }

    free(a);
    return;
}
